#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;

    while (t--) {
        int n;
        cin >> n;

        vector<int> a(n);

        for (int i = 0; i < n; i++)
            cin >> a[i];

        // Process towers from left to right
        for (int i = 0; i < n; i++) {

            for (int j = i + 1; j < n; j++) {

                // First taller tower
                if (a[j] > a[i]) {
                    a[j] = a[i];
                    break;
                }
            }
        }

        long long ans = 0;

        for (int x : a)
            ans += x;

        cout << ans << '\n';
    }

    return 0;
}
